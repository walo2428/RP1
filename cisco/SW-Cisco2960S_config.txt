! ============================================================
! Switch Cisco Catalyst 2960-S — SW-IRIS
! Infrastructure IRIS Nice — RP-01
! Talibi Omar — BTS SIO SISR — Session 2026
! IP Management : 192.168.50.253
! ============================================================

hostname SW-IRIS

! VLANs
vlan 10
 name IRIS-Etudiants
vlan 20
 name Professeurs
vlan 30
 name Administration
vlan 40
 name Guest
vlan 50
 name Management
vlan 99
 name PRE-AUTH

! AAA + 802.1X global
aaa new-model
aaa authentication dot1x default group radius
aaa authorization network default group radius
dot1x system-auth-control

! Serveur RADIUS (SRV-DC01)
radius-server host 192.168.50.10 auth-port 1812 acct-port 1813 key [SECRET_RADIUS]
radius-server timeout 5
radius-server retransmit 2

! Port uplink trunk vers R1-IRIS
interface GigabitEthernet0/1
 description Trunk-vers-R1-IRIS
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,50,99
 switchport trunk native vlan 99
 no shutdown

! Port vers SRV-DC01 (Management - pas de 802.1X)
interface GigabitEthernet0/24
 description SRV-DC01-Management
 switchport mode access
 switchport access vlan 50
 spanning-tree portfast
 no shutdown

! Port vers AP-IRIS (trunk pour VLANs WiFi)
interface GigabitEthernet0/23
 description AP-IRIS-C9105AXI
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40,99
 switchport trunk native vlan 99
 no shutdown

! Ports utilisateurs filaires avec 802.1X (VLAN 99 par defaut)
interface range FastEthernet0/1 - 20
 description Ports-Utilisateurs-802.1X
 switchport mode access
 switchport access vlan 99
 dot1x port-control auto
 dot1x host-mode multi-auth
 authentication event fail action next-method
 authentication event server dead action authorize vlan 99
 spanning-tree portfast
 no shutdown

! Interface Management VLAN 50
interface Vlan50
 description Management-IP
 ip address 192.168.50.253 255.255.255.0
 no shutdown

ip default-gateway 192.168.50.254

! Journaux syslog vers SRV-DC01
logging host 192.168.50.10
logging source-interface Vlan50
logging trap informational
service timestamps log datetime msec

! Supervision SNMP
snmp-server community iris-nice RO
snmp-server host 192.168.50.10 version 2c iris-nice

! Acces SSH uniquement
ip domain-name iris.local
crypto key generate rsa modulus 2048
ip ssh version 2
username admin privilege 15 secret [MOT_DE_PASSE_ADMIN]
line vty 0 15
 transport input ssh
 login local

! Desactiver les ports inutilises
interface range FastEthernet0/21 - 22
 shutdown

end
